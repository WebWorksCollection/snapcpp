project(plugin_password)

find_package( Qt4 4.8.1 QUIET REQUIRED QtCore )
include( ${QT_USE_FILE} )
add_definitions( ${QT_DEFINITIONS} )

include_directories(
    SYSTEM
        ${QT_INCLUDES}
)

include_directories(
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CONTROLLEDVARS_INCLUDE_DIRS}
)

QT4_ADD_RESOURCES(SNAP_MANAGER_RESOURCES_RCC
    password.qrc
)



snap_validate_xml( content.xml                          content/content.dtd )

snap_validate_xml( editor/password-blacklist-page.xml   editor/editor-form.dtd )
snap_validate_xml( editor/password-policy-page.xml      editor/editor-form.dtd )

snap_validate_xml( layout/password-blacklist-parser.xsl layout/simplified-xslt20.dtd )
snap_validate_xml( layout/password-policy-parser.xsl    layout/simplified-xslt20.dtd )
snap_validate_xml( password_widgets/password-form.xsl   layout/simplified-xslt20.dtd )

snap_validate_js( password.js )


add_library(password SHARED
    password.cpp
    password.h
    blacklist.cpp
    policy.cpp
    ${SNAP_MANAGER_MOC_CXX}
    ${SNAP_MANAGER_RESOURCES_RCC}
    content.xml
    editor/password-blacklist-page.xml
    editor/password-policy-page.xml
    layout/password-blacklist-parser.xsl
    layout/password-policy-parser.xsl
    password.css
    password.js
    password_widgets/password-form.xsl
)

target_link_libraries( password ${QT_LIBRARIES} )

get_property( 3RDPARTY_INCLUDED GLOBAL PROPERTY 3RDPARTY_INCLUDED )
if( 3RDPARTY_INCLUDED )
    add_dependencies( password make_controlled_vars_headers )
endif()

install(
    TARGETS password LIBRARY DESTINATION ${PLUGIN_INSTALL_DIR}
)
install(
    DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
    DESTINATION include/snapwebsites/plugins/password
    FILES_MATCHING PATTERN "*.h"
)
